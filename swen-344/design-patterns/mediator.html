Mediator Pattern Example 

$(document).ready(function(){
  var books = [new Book('Moby Dick', 'Herman Melville'),
              new Book("Gulliver's Travels", 'Jonathan Swift'),
              new Book('Pride and Prejudice', 'Jane Austen'),
              new Book('To Kill a Mockingbird', 'Harper Lee')];
  var mediator = new Mediator($('#loading'), $('#list'), books);
  $('#search').click(function(){
    mediator.search($('#search_input').val());
  });
});

var Mediator = function(loading, list, books) {
    this.list = list;
    this.books = books;
    this.init()
}

  Mediator.prototype = {
      init: function() {
        // append items to list
      },
      search: function(keyword) {
        var array = [];
        for (x in this.books){
          if(this.books[x].search(keyword))
            array.push(this.books[x]);
        }
        this.set_result(array);
      },
      set_result: function(array) {
        //append items to list
      }
  }

var Book = function(name, author) {
    this.name = name;
    this.author = author;
    this.search = function(keyword) {
        return !(this.name.indexOf(keyword) == -1 && this.author.indexOf(keyword) == -1);
    }
}
